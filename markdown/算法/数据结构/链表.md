# 单链表
- 面试时候常用链表，但是笔试一般用数组模拟
- 一般用于存储邻接表,存储树和图
- 用$e[N]$ 来存储接节点数值，用$ne[N]$来存储下一个节点的位置
- 用数组模拟链表也称为静态链表



```python
n=int(input())
N=100010
e=[0]*N
ne=[0]*N
idx=1
head=-1
def add_to_head(x):
    global idx,head
    e[idx]=x
    ne[idx]=head
    head=idx
    idx+=1
def insert(k,x):
    global idx
    e[idx]=x
    ne[idx]=ne[k]
    ne[k]=idx
    idx+=1
def delete(k):
    ne[k]=ne[ne[k]]
for i in range(n):
    op=list(input().split())
    if op[0]=='H':
        add_to_head(int(op[1]))
    elif op[0]=='I':
        insert(int(op[1]),int(op[2]))
    elif op[0]=='D':
        if op[1]=='0':
            head=ne[head]
        delete(int(op[1]))

while head!=-1:
    print(e[head],end=' ')
    head=ne[head]
```


# 双链表
- 优化某些问题


```python
n=int(input())
N=100010
e=[0]*N
ne=[0]*N
pe=[0]*N
head=0
idx=1

def add_to_head(x):
    global idx
    e[idx]=x
    if ne[head]:
        ne[idx]=ne[head]
        pe[ne[head]]=idx
        pe[idx]=head
        ne[head]=idx
    else:
        ne[idx]=head
        pe[idx]=head
        pe[head]=idx
        ne[head]=idx
    idx+=1
        
def insert_L(k,x):
    global idx
    e[idx]=x
    pe[idx]=pe[k]
    ne[pe[k]]=idx
    ne[idx]=k
    pe[k]=idx
    idx+=1
    
def insert_R(k,x):
    global idx
    e[idx]=x
    ne[idx]=ne[k]
    pe[ne[k]]=idx
    ne[k]=idx
    pe[idx]=k
    idx+=1
def add_to_tail(x):
    global idx
    if ne[head]==0:
        add_to_head(x)
        return
    e[idx]=x
    pe[idx]=pe[head]
    ne[pe[head]]=idx
    pe[head]=idx
    ne[idx]=head
    idx+=1
    

def delete(k):
    pe[ne[k]]=pe[k]
    ne[pe[k]]=ne[k]

for i in range(n):
    p=list(input().split())
    if p[0]=='L':
        add_to_head(int(p[1]))
    elif p[0]=='R':
        add_to_tail(int(p[1]))
    elif p[0]=='IR':
        insert_R(int(p[1]),int(p[2]))
    elif p[0]=='IL':
        insert_L(int(p[1]),int(p[2]))
    elif p[0]=='D':
        delete(int(p[1]))
cur=ne[head]
while cur!=0:
    print(e[cur],end=' ')
    cur=ne[cur]
    
```
