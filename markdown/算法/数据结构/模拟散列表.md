1、拉链法代码

```python
n = int(input())
N = 100003
h = [-1]*(N)
e = [0]*(N)
ne = [0]*N
idx = 0
def insert(x):
    global idx
    k = (x % N + N) % N
    e[idx] = x
    ne[idx] = h[k]
    h[k] = idx
    idx += 1

def find(x):
    k = (x % N + N) % N
    cur = h[k]
    while cur != -1:
        if e[cur] == x:
            return True
        cur = ne[cur]
    return False
        
for i in range(n):
    p = list(input().split())
    if p[0] == 'I':
        insert(int(p[1]))
    else:
        if find(int(p[1])):
            print('Yes')
        else:
            print('No')

```

2、开放寻址法代码

```python
n = int(input())
N = 200003
h = [None] * N

def insert(x):
    k = (x % N + N) % N
    while h[k] != None:
        k += 1
        if k ==N:
            k = 0
    h[k] = x

def find(x):
    k = (x % N + N) % N
    while h[k] != None:
        if h[k] == x:
            return True
        k += 1
        if k == N:
            k = 0
    return False
    
        
for i in range(n):
    p = list(input().split())
    if p[0] == 'I':
        insert(int(p[1]))
    else:

            print('No')
```
