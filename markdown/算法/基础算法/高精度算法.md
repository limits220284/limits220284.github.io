# 高精度加法

```python
a = [int(x) for x in input()]
b = [int(x) for x in input()]
m, n = len(a), len(b)
i = 0 
a.reverse()
b.reverse()
cnt = 0
tot = []
while i < m or i < n:
    if i < m: cnt += a[i]
    if i < n: cnt += b[i]
    tot.append(cnt%10)
    cnt = cnt//10
    i += 1
if cnt:
    tot.append(cnt)
tot.reverse()
print(''.join(str(x) for x in tot))
```

# 高精度减法

```python
a = [int(x) for x in input()]
b = [int(x) for x in input()]
m, n = len(a), len(b)
def cmp(a, b):
    if len(a) != len(b):
        return len(a) > len(b)
    for i in range(len(a)):
        if a[i] != b[i]:
            return a[i] > b[i]
    return True
#保证a是最大的即可
sign = ''
if not cmp(a, b):
    a, b = b, a
    sign = '-'
a.reverse()
b.reverse()
m, n = len(a), len(b)
i = 0
t = 0
tot = []
while i < m:
    t = a[i] - t
    if i < n: t -= b[i]
    tot.append(t%10)
    t = 1 if t < 0 else 0
    i += 1

while len(tot) > 1 and tot[-1] == 0:
    tot.pop()
tot.reverse()
print(sign + ''.join(str(x) for x in tot))
```

# 高精度乘法

```python
a = list(int(x) for x in input())
b = list(int(x) for x in input())


#主要分两步,一步是计算每一位的数值，另一步骤是处理进位
m, n = len(a), len(b)
a.reverse()
b.reverse()
res=[0] * (m + n)
for i in range(m):
    for j in range(n):
        res[i+j] += a[i] * b[j]
tot = 0
for i in range(m+n):
    tot = res[i] + tot
    res[i] = tot % 10
    tot //= 10
while len(res) > 1 and res[-1] == 0:
    res.pop()
res.reverse()
print(''.join(str(x) for x in res))
```
