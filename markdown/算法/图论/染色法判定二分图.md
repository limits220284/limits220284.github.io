## 染色法判定二分图
- 一个图是二分图 $\Leftrightarrow$ 该图可以被二染色,不含奇数环

### 深度优先搜索

```python
n,m=map(int,input().split())
N=100010
M=200010
h=[-1]*N
e=[0]*M
ne=[0]*M
idx=0
color=[0]*(n+1)

def add(a,b):
    global idx
    e[idx]=b
    ne[idx]=h[a]
    h[a]=idx
    idx+=1

def dfs(i,c):
    color[i]=c
    cur=h[i]
    while cur!=-1:
        j=e[cur]
        if color[j]==0:
            if dfs(j,3-c)==False:
                return False
        elif color[j]==c:
            return False
        cur=ne[cur]
    return True


for i in range(m):
    a,b=map(int,input().split())
    add(a,b)
    add(b,a)
flag=True
for i in range(1,n+1):
    if color[i]==0:
        if dfs(i,1)==False:
            flag=False
            break
if flag:
    print('Yes')
else:
    print('No')
```


### 广度优先搜索


```python
import collections
from collections import deque
n,m=map(int,input().split())
N=100010
M=200010
h=[-1]*N
e=[0]*M
ne=[0]*M
idx=0
color=[0]*(n+1)

def add(a,b):
    global idx
    e[idx]=b
    ne[idx]=h[a]
    h[a]=idx
    idx+=1

def bfs(i):
    que=deque()
    que.append(i)
    color[i]=1
    while que:
        t=que.popleft()
        nc=3-color[t]
        cur=h[t]
        while cur!=-1:
            j=e[cur]
            if color[j]==0:
                color[j]=nc
                que.append(j)
            elif color[j]==color[t]:
                return False
            cur=ne[cur]

    return True


for i in range(m):
    a,b=map(int,input().split())
    add(a,b)
    add(b,a)
flag=True
for i in range(1,n+1):
    if color[i]==0 and bfs(i)==False:
            flag=False
            break
if flag:
    print('Yes')
else:
    print('No')

```
